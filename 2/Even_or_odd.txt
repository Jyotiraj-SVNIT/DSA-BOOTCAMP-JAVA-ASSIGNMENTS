import java.util.Scanner;

public class MyClass {
    public static void main(String args[]) {
        Scanner in=new Scanner(System.in);
        int num=in.nextInt();
        if(num%2==0){
            System.out.println("So the number you typed i.e "+num+" is even");
        }else{
            System.out.println("So the number you typed i.e "+num+" is odd");
        }
    }
}
